#!/bin/sh

echo "Running Python checks..."

FILES=$(git diff --cached --name-only --diff-filter=ACMRT | grep '\.py$')

if [ -z "$FILES" ]; then
  echo "No Python files to check."
  exit 0
fi

FAILED=0

# Syntax check
for file in $FILES; do
  python -m py_compile "$file" || FAILED=1
done

if [ $FAILED -ne 0 ]; then
  echo "Checks failed. Commit aborted."
  exit 1
fi


# Linting
ruff check $FILES || FAILED=1

if [ $FAILED -ne 0 ]; then
  echo "Checks failed. Commit aborted."
  exit 1
fi

# echo "All checks passed."
exit 0
